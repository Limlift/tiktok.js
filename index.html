Tiktok.js

使用引导

1.引入JQuery

<script type="text/javascript" src="http://s2.dayoo.com/jq/1.7.2/jquery-1.7.2.min.js"></script>

要求jQuery 1.7+版本


2.引入Tiktok.js插件代码

<script type="text/javascript" src="/path/tiktok.js"></script>
<link rel="stylesheet" href="tiktok.css" />

3.调用Tiktok.js

$(element).tiktok();

默认设置

jQuery(element).tiktok({
	animateSpeed: 400, //动画速度
	delay: 200, //动画延时
	themeColor  :'#0082e5',//主颜色十六进制
	titleColor : '#0082e5',//标题栏颜色十六进制
	status : 'default',//浮动提示类型：done完成类型，alert警告类型，error错误类型，default默认类型
	position : null,//提示位置，适用于type为floattip（有上下左右）和poptip(有上中下)
	type : 'popbox',//popbox、poptext、popload、floattip、poptip提示类型
	content : {
		'title':'tip_title',//提示标题
		'content':'tip_content',//提示内容
		'picurl':null//提示加载图片
	},
	buttonCancel: '取消', //取消按钮文字
	buttonConfirm: '确定', //确定文字按钮文字
	headerAlign : 'center',//标题对齐方式
	contentAlign : 'left',//内容对齐方式
	ajaxContentUrl: null, //ajax请求地址
	onConfirm:null,//确认按钮按下后执行函数
	onCancel:null,//取消按钮按下后执行函数
	onBeforeShow: null,//显示之前执行函数
	onShow: null,//显示之时执行函数
	onHide: null//隐藏之后执行函数
})

显示隐藏方法

悬浮提示显示隐藏删除
$('#tt-sh').tiktok({
			type: 'floattip',
			position:'top',
			content : {
			'content':'原始文字提示'
			}
});
$('#tt-showbox').on('click',function() {
		$('#tt-sh').tiktok('show');
});
$('#tt-hidebox').on('click',function() {
		$('#tt-sh').tiktok('hide');
});
$('#tt-delbox').on('click',function() {
		$('#tt-sh').tiktok('destroy');
});

内容更新方法

内容更新
$('#tt-update').tiktok({
			type: 'floattip',
			position:'top',
			content : {
			'content':'原始文字提示'
			}
});
$('#tt-updatebox').bind('change',function() {
			$('#tt-update').tiktok('update','content',{content:$('#tt-updatebox').val()});
});
